int vlvPin = 13; //VALVULA no pin 13
int inputPin = 2; //infrared
int val = 0; //variavel do senso
int tempo; //variÃ¡vel de delay
void setup()
{
pinMode(vlvPin, OUTPUT); //declara a valvula as output
pinMode(inputPin, INPUT); //declare infrared sensor as input
}
 
void loop(){
 
val = digitalRead(inputPin); // read input value
 
if (val == HIGH) { //check if the input is HIGH
digitalWrite(vlvPin, HIGH); //VALVULA off
}
else if(val==LOW) {
for(tempo=420; tempo>=0; tempo --){
delay(1000);
if(tempo==0){
digitalWrite(vlvPin, HIGH); //VALVULA turned on
}
}
}
